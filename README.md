# Check-even-or-odd-number-in-java
Check even or odd number in java
public class KiemTraSoChanLe {
     public static void main(String[] args) {
         int n = 10;
         if(n%2==0)
         {
             System.out.println("This is an even number!");
         }
         else
         {
             System.out.println("This is an odd number!");
         }
     }
}
